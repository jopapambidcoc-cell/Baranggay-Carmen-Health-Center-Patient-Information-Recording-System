    <script>
        function clearPatientForm() {
            const form = document.getElementById('patientForm');
            
            // Wait 100ms so the CSS modal triggers before the fields disappear
            setTimeout(() => {
                form.reset();
                
                // Manually uncheck the illness toggle so the extra input hides
                const illnessToggle = document.getElementById('illness-toggle');
                if (illnessToggle) {
                    illnessToggle.checked = false;
                }
            }, 100);
        }
