    <script>
        function clearUpdateForm() {
            const form = document.getElementById('updateForm');
            setTimeout(() => {
                form.reset();
                document.getElementById('recordSelect').value = "";
                document.getElementById('diagInput').value = "";
                document.getElementById('prescInput').value = "";
                document.getElementById('notesInput').value = "";
            }, 100);
        }

        // NEW DELETE FUNCTION
        function deletePatientRecord(rowId, name) {
            // This creates a pop-up confirmation
            const confirmDelete = confirm("Are you sure you want to delete the record for " + name + "?");
            
            if (confirmDelete) {
                const row = document.getElementById(rowId);
                row.style.transition = "0.5s";
                row.style.opacity = "0"; // Fades it out
                
                setTimeout(() => {
                    row.remove(); // Removes it from the current view
                }, 500);
            }
        }
    </script>
